WITH VENDAS AS (

    SELECT
        CPF_CNPJ,
        VENDA_ID,
        CAST(INCLUSAO_DATA AS DATE) AS DT_COMPRA,
        DATE_TRUNC('MONTH', INCLUSAO_DATA) AS MES,
        SUM(VALOR_TOTAL) AS RECEITA
    FROM FSJ_PRD.GOLD.VW_ANALISE_VENDAS
    WHERE IDENTIFICADOR = 'V'
      AND CPF_CNPJ IS NOT NULL
    GROUP BY
        CPF_CNPJ,
        VENDA_ID,
        CAST(INCLUSAO_DATA AS DATE),
        DATE_TRUNC('MONTH', INCLUSAO_DATA)

),

PRIMEIRA_COMPRA AS (

    SELECT
        CPF_CNPJ,
        MIN(DT_COMPRA) AS DT_PRIMEIRA_COMPRA
    FROM VENDAS
    GROUP BY CPF_CNPJ

)

SELECT

    V.MES,

    COUNT(DISTINCT CASE
        WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) = V.MES
        THEN V.CPF_CNPJ
    END) AS NOVOS_CLIENTES,

    COUNT(DISTINCT CASE
        WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) < V.MES
        THEN V.CPF_CNPJ
    END) AS CLIENTES_RECORRENTES,

    SUM(CASE
        WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) = V.MES
        THEN V.RECEITA
        ELSE 0
    END) AS RECEITA_NOVOS,

    SUM(CASE
        WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) < V.MES
        THEN V.RECEITA
        ELSE 0
    END) AS RECEITA_RECORRENTES,

    ROUND(
        COUNT(DISTINCT CASE
            WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) = V.MES
            THEN V.CPF_CNPJ
        END)
        * 100.0
        / COUNT(DISTINCT V.CPF_CNPJ),
        2
    ) AS SHARE_NOVOS_CLIENTES,

    ROUND(
        COUNT(DISTINCT CASE
            WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) < V.MES
            THEN V.CPF_CNPJ
        END)
        * 100.0
        / COUNT(DISTINCT V.CPF_CNPJ),
        2
    ) AS SHARE_CLIENTES_RECORRENTES,

    ROUND(
        SUM(CASE
            WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) = V.MES
            THEN V.RECEITA
            ELSE 0
        END)
        * 100.0
        / NULLIF(SUM(V.RECEITA), 0),
        2
    ) AS SHARE_RECEITA_NOVOS,

    ROUND(
        SUM(CASE
            WHEN DATE_TRUNC('MONTH', P.DT_PRIMEIRA_COMPRA) < V.MES
            THEN V.RECEITA
            ELSE 0
        END)
        * 100.0
        / NULLIF(SUM(V.RECEITA), 0),
        2
    ) AS SHARE_RECEITA_RECORRENTES

FROM VENDAS V

INNER JOIN PRIMEIRA_COMPRA P
    ON V.CPF_CNPJ = P.CPF_CNPJ

GROUP BY
    V.MES

ORDER BY
    V.MES;
