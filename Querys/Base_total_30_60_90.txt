WITH CLIENTES AS (

    SELECT
        CPF_CNPJ,
        TO_DATE(DATE_TRUNC('MONTH', DATA_INCLUSAO)) AS ANO_MES,
        STATUS,
        SEGMENTO
    FROM FSJ_PRD.GOLD.DATAMART_CLIENTES
    WHERE TIPO_PESSOA = 'CPF_VALIDO'

),

ULTIMA_COMPRA AS (

    SELECT
        CPF_CNPJ,
        MAX(CAST(INCLUSAO_DATA AS DATE)) AS DT_ULTIMA_COMPRA
    FROM FSJ_PRD.GOLD.VW_ANALISE_VENDAS
    WHERE IDENTIFICADOR = 'V'
    GROUP BY CPF_CNPJ

)

SELECT

    C.ANO_MES,
    C.STATUS,
    C.SEGMENTO,

    COUNT(DISTINCT C.CPF_CNPJ) AS QTD_CLIENTES,

    COUNT(DISTINCT CASE
        WHEN U.DT_ULTIMA_COMPRA >= DATEADD(DAY,-29,CURRENT_DATE)
        THEN C.CPF_CNPJ
    END) AS CLIENTES_ATIVOS_30_DIAS,

    COUNT(DISTINCT CASE
        WHEN U.DT_ULTIMA_COMPRA >= DATEADD(DAY,-59,CURRENT_DATE)
        THEN C.CPF_CNPJ
    END) AS CLIENTES_ATIVOS_60_DIAS,

    COUNT(DISTINCT CASE
        WHEN U.DT_ULTIMA_COMPRA >= DATEADD(DAY,-89,CURRENT_DATE)
        THEN C.CPF_CNPJ
    END) AS CLIENTES_ATIVOS_90_DIAS

FROM CLIENTES C

LEFT JOIN ULTIMA_COMPRA U
       ON C.CPF_CNPJ = U.CPF_CNPJ

GROUP BY
    C.ANO_MES,
    C.STATUS,
    C.SEGMENTO

ORDER BY
    C.ANO_MES,
    C.STATUS,
    C.SEGMENTO;
